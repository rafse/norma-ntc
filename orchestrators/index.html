    Orchestratori — Diagrammi di flusso¶

    Ogni orchestratore (Layer 2) chiama una sequenza di primitivi NTC (Layer 1) e restituisce list[CheckResult].

    check_rc_beam_slu — Trave c.a. SLU ✅¶

    flowchart TD
    IN["Input
    ─────────────────
    b, d, A_s, f_yk
    strength_class
    M_Ed, V_Ed
    Asw? s? cot_theta?"]

    IN --> MAT["concrete_strength_class
    → f_ck, f_ctm
    §4.1.2.1.1 Tab.4.1.I"]

    MAT --> FCD["concrete_design_compressive_strength
    → f_cd
    §4.1.2.1.1.1"]

    MAT --> FYD["steel_design_strength
    → f_yd
    §4.1.2.1.1.3"]

    FCD --> MRD["concrete_bending_resistance
    → M_Rd
    §4.1.2.3.1"]
    FYD --> MRD

    MRD --> CR1["✦ CheckResult
    Flessione SLU
    ratio = M_Ed / M_Rd
    §4.1.2.3.1"]

    FCD --> VBR{"Asw fornito?"}
    FYD --> VBR

    VBR -- "Sì (con staffe)" --> VRS["shear_resistance_with_stirrups
    → V_Rd
    §4.1.2.3.5.2"]

    VBR -- "No (senza staffe)" --> VRN["shear_resistance_no_stirrups
    → V_Rd
    §4.1.2.3.5.1"]

    VRS --> CR2["✦ CheckResult
    Taglio SLU
    ratio = V_Ed / V_Rd"]
    VRN --> CR2

    MAT --> ASMIN["concrete_beam_min_reinforcement
    → A_s,min
    §4.1.6.1.1"]

    ASMIN --> CR3["✦ CheckResult
    Armatura minima
    ratio = A_s,min / A_s
    §4.1.6.1.1"]

    CR1 --> OUT["list[CheckResult]"]
    CR2 --> OUT
    CR3 --> OUT

    check_rc_beam_sle — Trave c.a. SLE ⬜¶

    flowchart TD
    IN["Input
    ─────────────────
    b, d, A_s, f_yk
    strength_class
    sigma_c, sigma_s
    exposure_class
    sigma_s_ser, E_s
    rho_eff, phi, c"]

    IN --> MAT["concrete_strength_class
    → f_ck, f_ctm
    §4.1.2.1.1"]

    MAT --> SLC["concrete_stress_limit
    → sigma_c,max
    §4.1.2.2.5.1"]

    SLC --> CR1["✦ CheckResult
    Tensione CLS SLE
    ratio = sigma_c / sigma_c,max
    §4.1.2.2.5.1"]

    IN --> SLS["steel_stress_limit
    → sigma_s,max
    §4.1.2.2.5.2"]

    SLS --> CR2["✦ CheckResult
    Tensione acciaio SLE
    ratio = sigma_s / sigma_s,max
    §4.1.2.2.5.2"]

    IN --> WLI["concrete_crack_width_limit
    → w_max
    §4.1.2.2.4.4 Tab.4.1.IV"]

    IN --> EPS["concrete_crack_mean_strain
    → ε_am-cm
    §4.1.2.2.4.5"]

    IN --> SRM["concrete_crack_spacing
    → s_r,max
    §4.1.2.2.4.5"]

    EPS --> WK["concrete_crack_width
    → w_k
    §4.1.2.2.4.5"]
    SRM --> WK

    WK --> CR3["✦ CheckResult
    Fessurazione SLE
    ratio = w_k / w_max
    §4.1.2.2.4"]

    WLI --> CR3

    CR1 --> OUT["list[CheckResult]"]
    CR2 --> OUT
    CR3 --> OUT

    check_rc_column_slu — Pilastro c.a. SLU ⬜¶

    flowchart TD
    IN["Input
    ─────────────────
    b, h, A_s, f_yk
    strength_class
    N_Ed, M_Edy, M_Edz
    V_Ed, l_0, i
    Asw?, s?, cot_theta?"]

    IN --> MAT["concrete_strength_class
    → f_ck
    §4.1.2.1.1"]

    MAT --> FCD["concrete_design_compressive_strength
    → f_cd"]
    MAT --> FYD["steel_design_strength
    → f_yd"]

    IN --> LAM["concrete_slenderness
    → λ = l_0 / i
    §4.1.2.3.9.2"]

    FCD --> LAMLIM["concrete_slenderness_limit
    → λ_lim = 25/√v
    §4.1.2.3.9.2"]

    LAM --> CR1["✦ CheckResult
    Snellezza
    ratio = λ / λ_lim
    §4.1.2.3.9.2"]
    LAMLIM --> CR1

    FCD --> NRD["N_Rd = f_cd · A_c + f_yd · A_s"]
    FYD --> NRD

    FCD --> MRD["M_Rd per asse y e z
    (pressoflessione)"]
    FYD --> MRD

    NRD --> BIX["biaxial_bending_check
    → ratio biassiale
    §4.1.2.3.4.2"]
    MRD --> BIX

    BIX --> CR2["✦ CheckResult
    Pressoflessione deviata
    §4.1.2.3.4.2"]

    FCD --> VRD{"Asw fornito?"}
    FYD --> VRD

    VRD -- Sì --> VRS["shear_resistance_with_stirrups
    → V_Rd §4.1.2.3.5.2"]
    VRD -- No --> VRN["shear_resistance_no_stirrups
    → V_Rd §4.1.2.3.5.1"]

    VRS --> CR3["✦ CheckResult
    Taglio SLU"]
    VRN --> CR3

    MAT --> ACMIN["concrete_column_min_reinforcement
    → A_s,min
    §4.1.6.1.2"]

    ACMIN --> CR4["✦ CheckResult
    Armatura minima
    §4.1.6.1.2"]

    CR1 --> OUT["list[CheckResult]"]
    CR2 --> OUT
    CR3 --> OUT
    CR4 --> OUT

    check_steel_beam_slu — Trave acciaio SLU ⬜¶

    flowchart TD
    IN["Input
    ─────────────────
    grade, section_class
    W_pl, A, A_v, i_z
    l_LT, chi_LT_curve
    M_Ed, V_Ed, N_Ed"]

    IN --> GP["steel_grade_properties
    → f_yk, f_tk
    §4.2.1"]

    GP --> MRD["steel_bending_resistance
    → M_pl,Rd
    §4.2.11"]

    MRD --> CR1["✦ CheckResult
    Flessione SLU
    ratio = M_Ed / M_pl,Rd
    §4.2.11"]

    GP --> VRD["steel_shear_resistance
    → V_pl,Rd
    §4.2.18"]

    VRD --> VRED{"V_Ed > 0.5 V_pl,Rd?"}

    VRED -- Sì --> VSHEAR["steel_bending_shear_reduction
    → M_V,Rd ridotto
    §4.2.22"]

    VRED -- No --> CR2["✦ CheckResult
    Taglio SLU
    ratio = V_Ed / V_pl,Rd
    §4.2.18"]

    VSHEAR --> CR2

    GP --> LTBUCK["steel_lt_buckling_resistance
    → M_b,Rd
    §4.2.12"]

    LTBUCK --> CR3["✦ CheckResult
    Instabilità laterale
    ratio = M_Ed / M_b,Rd
    §4.2.12"]

    CR1 --> OUT["list[CheckResult]"]
    CR2 --> OUT
    CR3 --> OUT
